  color: white;

  color: black;
  border-color: black;

<template>
  <span class='SignInBtn' v-bind:class='{ "light": isLightTheme, "dark": isDarkTheme }'>
    <Button type='ghost' @click='handleClick'>
      Sign In
    </Button>
  </span>
</template>
 
<script>
 
export default {
  name: 'SignInBtn',
  props: {
    theme: {
      type: String,
      required: false,
      default: 'dark',
      validator: (val) => {
        return ['dark', 'light'].indexOf(val) !== -1;
      }
    }
  },
  computed: {
    isLightTheme() {
      return this.theme == 'light'
    },
    isDarkTheme() {
      return this.theme == 'dark'
    }
  },
  methods: {
    handleClick() {
      this.$router.push({ path: 'sign-in' });
    }
  }
}
</script>
